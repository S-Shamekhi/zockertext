int run_container(struct config cfg) {
    pid_t pid;

    /* Create PID, mount, and UTS namespaces so the container is isolated */
    if (unshare(CLONE_NEWPID | CLONE_NEWNS | CLONE_NEWUTS) < 0) {
        perror("unshare");
        return 1;
    }

    pid = fork();
    if (pid < 0) {
        perror("fork");
        return 1;
    }

    if (pid == 0) {
        /* Child: now inside new namespaces (PID will be 1 here) */

        /* Make mounts private so remounting /proc doesn't affect host */
        if (mount(NULL, "/", NULL, MS_REC | MS_PRIVATE, NULL) < 0) {
            perror("mount private");
            /* continue anyway */
        }

        /* Mount a fresh procfs for this PID namespace */
        if (mount("proc", "/proc", "proc", 0, NULL) < 0) {
            perror("mount /proc");
            /* If this fails, ps won't show expected container PIDs */
        }

        /* Set hostname to container name */
        if (sethostname(cfg.name, strnlen(cfg.name, sizeof(cfg.name))) < 0) {
            perror("sethostname");
        }

        printf("Running child with pid (inside new PID ns): %d\n", getpid());
        execl("/bin/sh", "sh", "-c", cfg.command, NULL);
        perror("execl");
        exit(1);
    } else {
        /* Parent: still in host namespace */
        printf("[Parent] Child created with real-pid: %d (still in host namespace)\n", pid);
        waitpid(pid, NULL, 0);
        printf("[Parent] Done.\n");
    }
    return 0;
}
