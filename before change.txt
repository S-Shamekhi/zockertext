
int run_container(struct config cfg) {
    pid_t pid;

    if (unshare(CLONE_NEWPID) != 0) {
    	fprintf(stderr, "[ERR] Failed to unshare(2).");
   		return 1;
    }

    pid = fork();
    if (pid < 0) {
        return 1;
    }
    if (pid == 0) {
        printf("Running child with pid: %d\n", getpid());
        execl("/bin/sh", "sh", "-c", cfg.command, NULL);
    } else {
        sleep(2);
        waitpid(pid, NULL, 0);
        printf("[Parent] Stoping...\n");
    }
    return 0;
}
